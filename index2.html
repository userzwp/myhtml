setInterval(()=>{ let hr = document.querySelector(".hr") let mn = document.querySelector(".mn") let sc = document.querySelector(".sc") let hh = new Date().getHours()*30 let mm = new Date().getMinutes()*6 let ss = new Date().getSeconds()*6 hr.style.transform = `rotateZ(${hh+(mm/12)}deg)` mn.style.transform = `rotateZ(${mm}deg)` sc.style.transform = `rotateZ(${ss}deg)` })
